

CREATE TABLE CM_REQUEST (
	REQUEST_ID NUMBER(10),
	REQUEST_UUID VARCHAR2(255),
	SERVICE_PATH VARCHAR2(255),
	SERVICE_HOSTNAME VARCHAR2(255),
	SERVICE_PORT VARCHAR2(10),
	REQUEST_BODY VARCHAR2(4000),
	REQUEST_HEADER VARCHAR2(4000),
	START_DATETIME TIMESTAMP,
	END_DATETIME TIMESTAMP,
	TECHNICAL_STATUS VARCHAR2(100),
	RESPONSE_BODY VARCHAR2(4000),
	RESPONSE_HEADER VARCHAR2(4000),
	RESPONSE_HTTP_CODE VARCHAR2(5),
	FUNCTIONAL_STATUS VARCHAR2(100)
)
;

CREATE TABLE CM_ACTION (
	ACTION_ID NUMBER(10),
	REQUEST_ID NUMBER(10),
	ACTION_NUM NUMBER(5),
	ACTION_TYPE VARCHAR2(255),
	ACTION_DESCRIPTION VARCHAR2(255),
	ACTION_DATE TIMESTAMP
)
;

CREATE TABLE CM_ERROR (
	ERROR_ID NUMBER(10),
	REQUEST_ID NUMBER(10),
	ERROR_COMPONENT VARCHAR2(255),
	ERROR_CODE VARCHAR2(100),
	ERROR_DESCRIPTION VARCHAR2(255),
	ERROR_STACKTRACE VARCHAR2(4000),
	ERROR_DATE TIMESTAMP
)
;

CREATE TABLE CM_CONTEXT (
	CONTEXT_ID NUMBER(10),
	REQUEST_ID NUMBER(10),
	CONTEXT_TYPE VARCHAR2(100), --Functionnal or Technical
	CONTEXT_NAME VARCHAR2(100),
	CONTEXT_VALUE VARCHAR2(1000),
	CONTEXT_DATE TIMESTAMP
)
;
